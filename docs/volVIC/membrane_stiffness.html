 13def make_coordinates_systems(
 14    ctrl_pts: np.ndarray[np.floating], dN_dxi: sps.spmatrix, dN_deta: sps.spmatrix
 15) -> tuple[
 16    np.ndarray[np.floating],
 17    np.ndarray[np.floating],
 18    np.ndarray[np.floating],
 19    np.ndarray[np.floating],
 20    np.ndarray[np.floating],
 21]:
 22    """
 23    Compute the covariant basis vectors and transformation components between covariant and contravariant
 24    bases.
 25
 26    Given `ctrl_pts` (control points in physical space) and the derivatives of the basis functions with
 27    respect to the isoparametric coordinates (`dN_dxi`, `dN_deta`), this function returns the covariant basis
 28    vectors (`A1`, `A2`) and the transformation components (`A11`, `A22`, `A12`) for each evaluation point.
 29
 30    Parameters
 31    ----------
 32    ctrl_pts : np.ndarray[np.floating]
 33        Array of control points in physical space, shaped as (`3`, `n_nodes_xi`, `n_nodes_eta`).
 34    dN_dxi : sps.spmatrix
 35        Sparse matrix of derivatives of basis functions with respect to the first isoparametric coordinate
 36        (`xi`). Shape: (`n_gauss`, `n_nodes`).
 37    dN_deta : sps.spmatrix
 38        Sparse matrix of derivatives of basis functions with respect to the second isoparametric coordinate
 39        (`eta`). Shape: (`n_gauss`, `n_nodes`).
 40
 41    Returns
 42    -------
 43    tuple[np.ndarray[np.floating], np.ndarray[np.floating], np.ndarray[np.floating], np.ndarray[np.floating], np.ndarray[np.floating]]
 44        Tuple containing:
 45        - `A1` (`np.ndarray[np.floating]`): Covariant basis vector 1 at each evaluation point.
 46        - `A2` (`np.ndarray[np.floating]`): Covariant basis vector 2 at each evaluation point.
 47        - `A11` (`np.ndarray[np.floating]`): First component of the covariant-to-contravariant transformation.
 48        - `A22` (`np.ndarray[np.floating]`): Second component of the covariant-to-contravariant transformation.
 49        - `A12` (`np.ndarray[np.floating]`): Third component of the covariant-to-contravariant transformation.
 50
 51    Notes
 52    -----
 53    - The transformation components are computed pointwise for each evaluation point in the isoparametric space.
 54    - The returned tuple has length 5, with each entry corresponding to an array of values at all evaluation
 55    points.
 56    """
 57    # Covariant coordinates system
 58    A1 = ctrl_pts.reshape((3, -1)) @ dN_dxi.T
 59    A2 = ctrl_pts.reshape((3, -1)) @ dN_deta.T
 60    # Covariant to contravariant transformation
 61    A11cov = (A1 * A1).sum(axis=0)
 62    A22cov = (A2 * A2).sum(axis=0)
 63    A12cov = (A1 * A2).sum(axis=0)
 64    det_inv = 1 / (A11cov * A22cov - A12cov * A12cov)
 65    A11 = A22cov * det_inv
 66    A22 = A11cov * det_inv
 67    A12 = -A12cov * det_inv
 68
 69    return A1, A2, A11, A22, A12

 72def make_H(
 73    A11: np.ndarray[np.floating],
 74    A22: np.ndarray[np.floating],
 75    A12: np.ndarray[np.floating],
 76) -> sps.spmatrix:
 77    """
 78    Assemble Hooke's tensor in contravariant basis Voigt notation for a membrane material with Poisson's ratio
 79    set to 0 and Young's modulus set to 1, using B-spline basis components.
 80
 81    Parameters
 82    ----------
 83    A11 : np.ndarray[np.floating]
 84        Covariant-to-contravariant basis component array for the (1,1) direction in isoparametric space.
 85    A22 : np.ndarray[np.floating]
 86        Covariant-to-contravariant basis component array for the (2,2) direction in isoparametric space.
 87    A12 : np.ndarray[np.floating]
 88        Covariant-to-contravariant basis component array for the (1,2) direction in isoparametric space.
 89
 90    Returns
 91    -------
 92    sps.spmatrix
 93        Hooke's tensor as a sparse matrix in contravariant basis Voigt notation, with shape
 94        (`3 * n_gauss`, `3 * n_gauss`), where `n_gauss` is the number of isoparametric points.
 95
 96    Notes
 97    -----
 98    - The tensor is block-assembled using `scipy.sparse.bmat` and diagonal matrices from the input arrays.
 99    - The material law is restricted to Poisson's ratio = 0 and Young's modulus = 1.
100    - The resulting tensor is suitable for use in isogeometric Kirchhoff-Love membrane formulations.
101    """
102    H11 = sps.diags(A11 * A11)
103    H22 = sps.diags(A22 * A22)
104    H33 = sps.diags(0.5 * (A11 * A22 + A12 * A12))
105    H12 = sps.diags(A12 * A12)
106    H13 = sps.diags(A11 * A12)
107    H23 = sps.diags(A22 * A12)
108    H = sps.bmat([[H11, H12, H13], [H12, H22, H23], [H13, H23, H33]])
109    return H

112def make_Bm(
113    dN_dxi: sps.spmatrix,
114    dN_deta: sps.spmatrix,
115    A1: np.ndarray[np.floating],
116    A2: np.ndarray[np.floating],
117) -> sps.spmatrix:
118    """
119    Assemble the Jacobian matrix `Bm` for membrane deformation in Voight notation in isogeometric analysis.
120
121    This function constructs the sparse matrix `Bm` by combining the derivatives of the shape functions in the
122    isoparametric (`xi`, `eta`) directions with the corresponding covariant tangent vectors `A1` and `A2`. The
123    resulting matrix maps nodal displacements to membrane strains in Voight notation, and is used in
124    isogeometric membrane finite element formulations.
125
126    Parameters
127    ----------
128    dN_dxi : sps.spmatrix
129        Sparse matrix of derivatives of the shape functions with respect to the `xi` coordinate in
130        isoparametric space. Shape: (`n_gauss`, `n_nodes`).
131    dN_deta : sps.spmatrix
132        Sparse matrix of derivatives of the shape functions with respect to the `eta` coordinate in
133        isoparametric space. Shape: (`n_gauss`, `n_nodes`).
134    A1 : np.ndarray[np.floating]
135        Covariant tangent vector in the `xi` direction. Shape: (`3`, `n_gauss`).
136    A2 : np.ndarray[np.floating]
137        Covariant tangent vector in the `eta` direction. Shape: (`3`, `n_gauss`).
138
139    Returns
140    -------
141    sps.spmatrix
142        Sparse matrix `Bm` representing the Jacobian of the membrane deformation transformation in Voight notation. Shape: (`3 * n_gauss`, `3 * n_nodes`).
143
144    Notes
145    -----
146    - The output matrix `Bm` has shape (`3 * n_gauss`, `3 * n_nodes`), where `n_gauss` is the number of
147    quadrature points and `n_nodes` is the number of control points (basis functions).
148    - The isoparametric space refers to the parametric domain of the B-spline basis.
149    """
150    de_xi_du = sps.hstack(
151        (
152            dN_dxi.multiply(A1[0, :, None]),
153            dN_dxi.multiply(A1[1, :, None]),
154            dN_dxi.multiply(A1[2, :, None]),
155        )
156    )
157    de_eta_du = sps.hstack(
158        (
159            dN_deta.multiply(A2[0, :, None]),
160            dN_deta.multiply(A2[1, :, None]),
161            dN_deta.multiply(A2[2, :, None]),
162        )
163    )
164    de_xi_eta_2_du = sps.hstack(
165        (
166            dN_deta.multiply(A1[0, :, None]) + dN_dxi.multiply(A2[0, :, None]),
167            dN_deta.multiply(A1[1, :, None]) + dN_dxi.multiply(A2[1, :, None]),
168            dN_deta.multiply(A1[2, :, None]) + dN_dxi.multiply(A2[2, :, None]),
169        )
170    )
171    Bm = sps.vstack((de_xi_du, de_eta_du, de_xi_eta_2_du))
172    return Bm

175def make_membrane_stiffness_operator(
176    spline: BSpline, ctrl_pts: np.ndarray[np.floating]
177) -> sps.spmatrix:
178    """
179    Assemble the global membrane stiffness matrix for a B-spline surface patch.
180
181    This function computes the global stiffness operator `K` for a B-spline surface patch, considering only
182    in-plane (membrane) strain energy contributions and omitting out-of-plane (bending) effects. The assembly
183    is performed using Gauss-Legendre quadrature over the isoparametric domain.
184
185    Parameters
186    ----------
187    spline : BSpline
188        B-spline surface patch object defining the geometry and basis functions.
189    ctrl_pts : np.ndarray[np.floating]
190        Array of control points defining the physical geometry of the surface.
191        Shape should be (`3`, `n_nodes_xi`, `n_nodes_eta`).
192
193    Returns
194    -------
195    K : sps.spmatrix
196        Global stiffness matrix (`sps.spmatrix`) containing only membrane (in-plane) contributions.
197
198    Notes
199    -----
200    - The integration is performed using quadrature points determined by the spline degrees in each
201    isoparametric direction.
202    - The returned matrix `K` does not include bending or out-of-plane stiffness terms.
203    """
204    # Integration space
205    p, q = spline.getDegrees()
206    (xi, eta), (dxi, deta) = spline.gauss_legendre_for_integration([p + 1, q + 1])
207    # Local covariant and contravariant bases
208    dN_dxi, dN_deta = spline.DN([xi, eta], 1)  # type: ignore
209    A1, A2, A11, A22, A12 = make_coordinates_systems(ctrl_pts, dN_dxi, dN_deta)
210    A3 = np.cross(A1, A2, axis=0)
211    detJ = np.linalg.norm(A3, axis=0)
212    A3 = A3 / detJ[None]
213    WdetJ = detJ * np.outer(dxi, deta).ravel()
214    # Hooke tensor in the contravariant basis in Voigt notation
215    H = make_H(A11, A22, A12)
216    # Membrane strain tensor in the contravariant basis in Voigt notation
217    Bm = make_Bm(dN_dxi, dN_deta, A1, A2)
218    # Linear elasticity operator
219    K = (Bm.multiply(np.hstack([WdetJ] * 3)[:, None])).T @ H @ Bm
220    return K

223def make_membrane_stifness(
224    mesh: Mesh, verbose: bool = True, disable_parallel: bool = False
225) -> sps.spmatrix:
226    """
227    Assemble the global membrane stiffness matrix for a full multipatch mesh.
228
229    This function computes the global membrane stiffness matrix by summing the contributions
230    from all patches of the mesh. Each patch is processed separately using the B-spline geometry
231    and control points of each patch.
232
233    Parameters
234    ----------
235    mesh : Mesh
236        Mesh object containing B-spline patches.
237    verbose : bool, optional
238        If True, displays a progress bar during assembly. Default is True.
239    disable_parallel : bool, optional
240        If True, disables parallel computation. Default is False.
241
242    Returns
243    -------
244    sps.spmatrix
245        Global membrane stiffness matrix, including all patches, ready for use in
246        isogeometric membrane finite element analysis. The size corresponds to `3 * n_dofs`,
247        where `n_dofs` is the total number of control points in the mesh.
248
249    Notes
250    -----
251    - The function internally calls `make_membrane_stiffness_operator` for each patch.
252    - Uses parallel computation if optimal when `disable_parallel` is False.
253    - The resulting matrix includes only in-plane (membrane) stiffness contributions.
254    """
255    if isinstance(mesh, MeshLattice):
256        cell_mesh = mesh.get_mesh_cell_one()
257        separated_ctrl_pts = cell_mesh.get_separated_ctrl_pts()
258        args_list = [
259            (cell_mesh.splines[patch], separated_ctrl_pts[patch])
260            for patch in range(cell_mesh.connectivity.nb_patchs)
261        ]
262        K_elem_cell = parallel_blocks(
263            make_membrane_stiffness_operator,
264            args_list,
265            verbose=verbose,
266            pbar_title="Make membrane stiffness",
267            disable_parallel=disable_parallel,
268        )
269        K_elem = K_elem_cell * (mesh.l * mesh.m * mesh.n)
270    else:
271        separated_ctrl_pts = mesh.get_separated_ctrl_pts()
272        args_list = [
273            (mesh.splines[patch], separated_ctrl_pts[patch])
274            for patch in range(mesh.connectivity.nb_patchs)
275        ]
276        K_elem = parallel_blocks(
277            make_membrane_stiffness_operator,
278            args_list,
279            verbose=verbose,
280            pbar_title="Make membrane stiffness",
281            disable_parallel=disable_parallel,
282        )
283    K = mesh.assemble_hessians(K_elem, 3)
284    return K

366def make_membrane_weight(
367    mesh: Mesh,
368    image_energies: Iterable[VirtualImageCorrelationEnergyElem],
369    membrane_K: sps.spmatrix,
370    rho: float = 1.5,
371    image_std: float = 5_000,
372    expected_mean_dist: float = 5,
373    n_intg: int = 100,
374) -> float:
375    """
376    Compute a membrane regularization weight by matching the expected membrane
377    energy to the expected virtual image correlation (VIC) energy.
378
379    The membrane energy is evaluated analytically from a probabilistic model
380    of the B-spline control point displacements, while the VIC energy is
381    estimated from finite differences of the virtual image response.
382
383    Parameters
384    ----------
385    mesh : Mesh
386        B-spline mesh defining the control points and basis functions.
387    image_energies : iterable of VirtualImageCorrelationEnergyElem
388        VIC energy objects associated with each patch.
389    membrane_K : scipy.sparse.spmatrix
390        Membrane stiffness matrix of size (3 * n_bf, 3 * n_bf).
391    rho : float, optional
392        Image correlation parameter passed to the virtual image operator. By default 1.5.
393    image_std : float, optional
394        Standard deviation of the image noise. By default 5_000.
395    expected_mean_dist : float, optional
396        Expected mean distance between the converged B-spline geometry and the
397        target image. By default 5.
398    n_intg : int, optional
399        Number of integration points used for estimating the VIC energy.
400        By default 100.
401
402    Returns
403    -------
404    membrane_weight : float
405        Regularization weight such that the expected membrane energy matches
406        the expected VIC energy.
407
408    Notes
409    -----
410    **Expected membrane energy**
411
412    Each B-spline control point ``a`` is assigned a random displacement
413
414        u_a = d_a * w_a * n_a    R^3
415
416    where:
417
418    - ``d_a = expected_mean_dist * (1 + _a)``, with ``_a ~ N(0, 1)``
419    - ``w_a = B_a(_a)`` is the B-spline basis evaluated at the Greville abscissa
420    - ``n_a`` is an isotropic random unit vector in R^3
421
422    The membrane stiffness matrix ``K  R^{3 n_bf  3 n_bf}`` is decomposed into
423    33 blocks ``K_{a,b}`` such that
424
425        [K_{a,b}]_{c,d} = K_{c * n_bf + a, d * n_bf + b}
426
427    The discrete membrane energy reads
428
429        E_mem(U) = 1/2 * _{a,b} w_a w_b d_a d_b n_a K_{a,b} n_b
430
431    Taking the expectation and assuming independence between radial amplitudes
432    and directions yields
433
434        E[E_mem] = 1/2 * _{a,b} w_a w_b E[d_a d_b] E[n_a K_{a,b} n_b]
435
436    For a  b, the isotropy of ``n_b`` implies
437
438        E[n_a K_{a,b} n_b] = 0
439
440    Hence only diagonal terms remain:
441
442        E[E_mem] = 1/2 * _a w_a E[d_a] E[n_a K_{a,a} n_a]
443
444    Using isotropy:
445
446        E[n_a n_a] = 1/3 * I_3
447         E[n_a K_{a,a} n_a] = 1/3 * tr(K_{a,a})
448
449    Moreover:
450
451        E[d_a] = expected_mean_dist * E[(1 + _a)] = 2 * expected_mean_dist
452
453    Finally:
454
455        E[E_mem] = expected_mean_dist / 3 * _a w_a * tr(K_{a,a})
456
457    Introducing:
458
459        W = (w_1, ..., w_{n_bf}, w_1, ..., w_{n_bf}, w_1, ..., w_{n_bf})
460
461    this can be written compactly as:
462
463        E[E_mem] = expected_mean_dist / 3 * (W  diag(K))
464
465    **Expected VIC energy**
466
467    The observed image near the converged surface is modeled as:
468
469        f() = g( + d) +  
470
471    where  ~ N(0,1),  = image_std, and d is a random normal offset along
472    the surface normal such that E(|d|) = expected_mean_dist.
473
474    The VIC energy is defined as
475
476        E_vic = 1/2 _ 1/(2h) _{-h}^{h} (f() - g()) d d
477
478    Substituting the model:
479
480        f() - g() = g( + d) - g() +  
481
482    and taking the expectation:
483
484        E[E_vic] = 1/2 _ 1/(2h) _{-h}^{h} E[(g( + d) - g())] d d
485                    + 1/2 _ 1/(2h) _{-h}^{h} E[ ] d d
486
487    The cross term vanishes because E[] = 0. The noise term evaluates to:
488
489        1/2 _ 1/(2h) _{-h}^{h}  d d = ||  / 2
490
491    Assuming the virtual image profile g is locally antisymmetric around the
492    surface (antisymmetry of the transition), the difference
493
494        g( + d) - g()
495
496    is symmetric with respect to -d/2, which implies the squared difference
497    depends only on |d|. As the law of d is otherwise unknown, the expectation
498    is approximated by evaluating the energy for a representative offset
499
500        d  expected_mean_dist
501
502    giving the final estimate:
503
504        E[E_vic]  ||/2 * (  + 1/(2h) _{-h}^{h} (g( + expected_mean_dist) - g()) d )
505
506    This is exactly what is implemented in the code: the integral is computed
507    by discrete summation on each patch.
508    """
509    weights = mesh.separated_to_unique(
510        [s.DN(s.greville_abscissa()).diagonal() for s in mesh.splines]
511    )
512    weights = np.hstack([weights] * 3)
513
514    E_mem = expected_mean_dist**2 / 3 * np.dot(weights * weights, membrane_K.diagonal())
515
516    z = np.zeros(1)
517    E_vic = 0
518    for e in image_energies:
519        gamma = np.linspace(-e.h, e.h, n_intg)
520        f = e.virtual_image(z, z, gamma + expected_mean_dist, rho)[0]
521        g = e.virtual_image(z, z, gamma, rho)[0]
522        mean_patch_error = np.mean((f - g) ** 2)
523        patch_area = e.wdetJs.sum()
524        E_vic += patch_area / 2 * (image_std**2 + mean_patch_error)
525
526    return E_vic / E_mem

176def make_membrane_stiffness_operator(
177    spline: BSpline, ctrl_pts: np.ndarray[np.floating]
178) -> sps.spmatrix:
179    """
180    Assemble the global membrane stiffness matrix for a B-spline surface patch.
181
182    This function computes the global stiffness operator `K` for a B-spline surface patch, considering only
183    in-plane (membrane) strain energy contributions and omitting out-of-plane (bending) effects. The assembly
184    is performed using Gauss-Legendre quadrature over the isoparametric domain.
185
186    Parameters
187    ----------
188    spline : BSpline
189        B-spline surface patch object defining the geometry and basis functions.
190    ctrl_pts : np.ndarray[np.floating]
191        Array of control points defining the physical geometry of the surface.
192        Shape should be (`3`, `n_nodes_xi`, `n_nodes_eta`).
193
194    Returns
195    -------
196    K : sps.spmatrix
197        Global stiffness matrix (`sps.spmatrix`) containing only membrane (in-plane) contributions.
198
199    Notes
200    -----
201    - The integration is performed using quadrature points determined by the spline degrees in each
202    isoparametric direction.
203    - The returned matrix `K` does not include bending or out-of-plane stiffness terms.
204    """
205    # Integration space
206    p, q = spline.getDegrees()
207    (xi, eta), (dxi, deta) = spline.gauss_legendre_for_integration([p + 1, q + 1])
208    # Local covariant and contravariant bases
209    dN_dxi, dN_deta = spline.DN([xi, eta], 1)  # type: ignore
210    A1, A2, A11, A22, A12 = make_coordinates_systems(ctrl_pts, dN_dxi, dN_deta)
211    A3 = np.cross(A1, A2, axis=0)
212    detJ = np.linalg.norm(A3, axis=0)
213    A3 = A3 / detJ[None]
214    WdetJ = detJ * np.outer(dxi, deta).ravel()
215    # Hooke tensor in the contravariant basis in Voigt notation
216    H = make_H(A11, A22, A12)
217    # Membrane strain tensor in the contravariant basis in Voigt notation
218    Bm = make_Bm(dN_dxi, dN_deta, A1, A2)
219    # Linear elasticity operator
220    K = (Bm.multiply(np.hstack([WdetJ] * 3)[:, None])).T @ H @ Bm
221    return K

377    Compute a membrane regularization weight by matching the expected membrane
378    energy to the expected virtual image correlation (VIC) energy.
379
380    The membrane energy is evaluated analytically from a probabilistic model
381    of the B-spline control point displacements, while the VIC energy is
382    estimated from finite differences of the virtual image response.
383
384    Parameters
385    ----------
386    mesh : Mesh
387        B-spline mesh defining the control points and basis functions.
388    image_energies : iterable of VirtualImageCorrelationEnergyElem
389        VIC energy objects associated with each patch.

391        Membrane stiffness matrix of size (3 * n_bf, 3 * n_bf).
392    rho : float, optional
393        Image correlation parameter passed to the virtual image operator. By default 1.5.
394    image_std : float, optional
395        Standard deviation of the image noise. By default 5_000.
396    expected_mean_dist : float, optional
397        Expected mean distance between the converged B-spline geometry and the
398        target image. By default 5.
399    n_intg : int, optional
400        Number of integration points used for estimating the VIC energy.
401        By default 100.
402
403    Returns
404    -------
405    membrane_weight : float
406        Regularization weight such that the expected membrane energy matches
407        the expected VIC energy.
408
409    Notes
410    -----
411    **Expected membrane energy**
412
413    Each B-spline control point ``a`` is assigned a random displacement
414
415        u_a = d_a * w_a * n_a    R^3
416
417    where:
418
419    - ``d_a = expected_mean_dist * (1 + _a)``, with ``_a ~ N(0, 1)``
420    - ``w_a = B_a(_a)`` is the B-spline basis evaluated at the Greville abscissa
421    - ``n_a`` is an isotropic random unit vector in R^3
422
423    The membrane stiffness matrix ``K  R^{3 n_bf  3 n_bf}`` is decomposed into
424    33 blocks ``K_{a,b}`` such that
425
426        [K_{a,b}]_{c,d} = K_{c * n_bf + a, d * n_bf + b}
427
428    The discrete membrane energy reads
429
430        E_mem(U) = 1/2 * _{a,b} w_a w_b d_a d_b n_a K_{a,b} n_b
431
432    Taking the expectation and assuming independence between radial amplitudes
433    and directions yields
434
435        E[E_mem] = 1/2 * _{a,b} w_a w_b E[d_a d_b] E[n_a K_{a,b} n_b]
436
437    For a  b, the isotropy of ``n_b`` implies
438
439        E[n_a K_{a,b} n_b] = 0
440
441    Hence only diagonal terms remain:
442
443        E[E_mem] = 1/2 * _a w_a E[d_a] E[n_a K_{a,a} n_a]
444
445    Using isotropy:
446
447        E[n_a n_a] = 1/3 * I_3
448         E[n_a K_{a,a} n_a] = 1/3 * tr(K_{a,a})
449
450    Moreover:
451
452        E[d_a] = expected_mean_dist * E[(1 + _a)] = 2 * expected_mean_dist
453
454    Finally:
455
456        E[E_mem] = expected_mean_dist / 3 * _a w_a * tr(K_{a,a})
457
458    Introducing:
459
460        W = (w_1, ..., w_{n_bf}, w_1, ..., w_{n_bf}, w_1, ..., w_{n_bf})
461
462    this can be written compactly as:
463
464        E[E_mem] = expected_mean_dist / 3 * (W  diag(K))
465
466    **Expected VIC energy**
467
468    The observed image near the converged surface is modeled as:
469
470        f() = g( + d) +  
471
472    where  ~ N(0,1),  = image_std, and d is a random normal offset along
473    the surface normal such that E(|d|) = expected_mean_dist.
474
475    The VIC energy is defined as
476
477        E_vic = 1/2 _ 1/(2h) _{-h}^{h} (f() - g()) d d
478
479    Substituting the model:
480
481        f() - g() = g( + d) - g() +  
482
483    and taking the expectation:
484
485        E[E_vic] = 1/2 _ 1/(2h) _{-h}^{h} E[(g( + d) - g())] d d
486                    + 1/2 _ 1/(2h) _{-h}^{h} E[ ] d d
487
488    The cross term vanishes because E[] = 0. The noise term evaluates to:
489
490        1/2 _ 1/(2h) _{-h}^{h}  d d = ||  / 2
491
492    Assuming the virtual image profile g is locally antisymmetric around the
493    surface (antisymmetry of the transition), the difference
494
495        g( + d) - g()
496
497    is symmetric with respect to -d/2, which implies the squared difference
498    depends only on |d|. As the law of d is otherwise unknown, the expectation
499    is approximated by evaluating the energy for a representative offset
500
501        d  expected_mean_dist
502
503    giving the final estimate:
504
505        E[E_vic]  ||/2 * (  + 1/(2h) _{-h}^{h} (g( + expected_mean_dist) - g()) d )
506
507    This is exactly what is implemented in the code: the integral is computed
508    by discrete summation on each patch.
509    """
510    weights = mesh.separated_to_unique(
511        [s.DN(s.greville_abscissa()).diagonal() for s in mesh.splines]
512    )
513    weights = np.hstack([weights] * 3)
514
515    E_mem = expected_mean_dist**2 / 3 * np.dot(weights * weights, membrane_K.diagonal())
516
517    z = np.zeros(1)
518    E_vic = 0
519    for e in image_energies:
520        gamma = np.linspace(-e.h, e.h, n_intg)
521        f = e.virtual_image(z, z, gamma + expected_mean_dist, rho)[0]
522        g = e.virtual_image(z, z, gamma, rho)[0]
523        mean_patch_error = np.mean((f - g) ** 2)
524        patch_area = e.wdetJs.sum()
525        E_vic += patch_area / 2 * (image_std**2 + mean_patch_error)
526
527    return E_vic / E_mem

  • w_a = B_a(_a) is the B-spline basis evaluated at the Greville abscissa
  • n_a is an isotropic random unit vector in R^3

The membrane stiffness matrix K R^{3 n_bf 3 n_bf} is decomposed into 33 blocks K_{a,b} such that

[K_{a,b}]_{c,d} = K_{c * n_bf + a, d * n_bf + b}

The discrete membrane energy reads

E_mem(U) = 1/2 * _{a,b} w_a w_b d_a d_b n_a K_{a,b} n_b

Taking the expectation and assuming independence between radial amplitudes and directions yields

E[E_mem] = 1/2 * _{a,b} w_a w_b E[d_a d_b] E[n_a K_{a,b} n_b]

For a b, the isotropy of n_b implies

E[n_a K_{a,b} n_b] = 0

Hence only diagonal terms remain:

E[E_mem] = 1/2 * _a w_a E[d_a] E[n_a K_{a,a} n_a]

Using isotropy:

E[n_a n_a] = 1/3 * I_3
 E[n_a K_{a,a} n_a] = 1/3 * tr(K_{a,a})

Moreover:

E[d_a] = expected_mean_dist * E[(1 + _a)] = 2 * expected_mean_dist

Finally:

E[E_mem] = expected_mean_dist / 3 * _a w_a * tr(K_{a,a})

Introducing:

W = (w_1, ..., w_{n_bf}, w_1, ..., w_{n_bf}, w_1, ..., w_{n_bf})

this can be written compactly as:

E[E_mem] = expected_mean_dist / 3 * (W  diag(K))
